-- Create database CREATE DATABASE librarydb; -- Create tables CREATE TABLE books ( id SERIAL PRIMARY KEY, title VARCHAR(100), author VARCHAR(100), available BOOLEAN DEFAULT true ); CREATE TABLE users ( id SERIAL PRIMARY KEY, name VARCHAR(50), email VARCHAR(100) ); CREATE TABLE borrow_records ( id SERIAL PRIMARY KEY, user_id INT REFERENCES users(id), book_id INT REFERENCES books(id), borrow_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP, return_date TIMESTAMP );
